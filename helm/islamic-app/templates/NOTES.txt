  export NODE_PORT=$(kubectl get --namespace {{ include "islamic-app.namespace" . }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "islamic-app.fullname" . }}-frontend)

  export SERVICE_IP=$(kubectl get svc --namespace {{ include "islamic-app.namespace" . }} {{ include "islamic-app.fullname" . }}-frontend --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.frontend.port }}

{{- else if contains "ClusterIP" .Values.service.frontend.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ include "islamic-app.namespace" . }} -l "app.kubernetes.io/name={{ include "islamic-app.name" . }},app.kubernetes.io/instance={{ .Release.Name }},app.kubernetes.io/component=frontend" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ include "islamic-app.namespace" . }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:3000 to use your application"
  kubectl --namespace {{ include "islamic-app.namespace" . }} port-forward $POD_NAME 3000:$CONTAINER_PORT
{{- end }}
